JavaScript Descriptor

shortcodes.register('carousel', {
anchor: '.shortcode-landing',
bind_fn: 'append',
template: 'section.carousel',
item_template: '.carousel-item',
item_anchor: '.slides-landing',
callback: function(template) {
$(template).find('.slides').slick({
  slidesToShow: 3,
  slidesToScroll: 1,
  dots: true,
  arrows: false,
  swipeToSlide: true
});
},
elements: {
img: {
  count: 'many',
  bind_fn: 'css',
  bind_property: 'background-image',
  extract_fn: 'attr',
  extract_attr: 'src',
  anchor: '.image-landing',
  anchor_element: 'item'
},
a: {
  count: 'many',
  anchor: '.link-landing',
  extract_fn: 'attr',
  extract_attr: ['href', 'title', 'html'],
  bind_fn: function(extract, $dest, props, descriptor, num) {
    if (num === 1) {
      $dest.find(props.anchor).append(''+extract[2]+'');
    } else if (num === 2) {
      $dest.find(props.anchor).append(''+extract[2]+'');
    } else {
      $dest.find(props.anchor).append(''+extract[2]+'');
    }
  },
  anchor_element: 'item'
},
h1: {
  count: 'many',
  bind_fn: 'html',
  anchor: '.title-landing',
  extract_fn: 'html',
  anchor_element: 'item'
},
rest: {
  anchor: '.text-landing',
  bind_fn: 'append',
  extract_fn: 'self',
  anchor_element: 'item'
}
}
});
