JavaScript Descriptor

shortcodes.register('carousel', {
  anchor: '.shortcode-landing',
  bind_fn: 'append',
  template: 'section.carousel',
  item_template: '.carousel-item',
  item_anchor: '.slides-landing',
  callback: function(template) {
    $('section.carousel').first().attr('id', 'main')
    $(template).find('.slides').not('.slick-initialized').slick({
      slidesToShow: 3,
        slidesToScroll: 1,
      dots: true,
      arrows: false,
      swipeToSlide: true,
      responsive: [
          {
            breakpoint: 960,
            settings: {
              slidesToShow: 1,
              slidesToScroll: 1
            }
          }
      ]
    });
  },
  elements: {
    img: {
      count: 'many',
      bind_extracted: {
        style_property: 'background-image',
        extracted_property: 'src'
      },
      extract: 'src',
      anchor: '.image-landing',
      anchor_element: 'item'
    },
    a: {
      count: 'many',
      anchor: '.link-landing',
      extract: ['href', 'target', 'html'],
      bind_extracted: function(dest, extract, props, descriptor, num) {
        if (num === 1) {
          dest.appendChild(parseDOM(`${extract.html}`))
        } else if (num === 2) {
          dest.appendChild(parseDOM(`${extract.html}`))
        } else {
          dest.appendChild(parseDOM(`${extract.html}`))
        }
      },
      anchor_element: 'item'
    },
    h1: {
      count: 'many',
      bind_extracted: 'html',
      anchor: '.title-landing',
      extract: 'html',
      anchor_element: 'item'
    },
    rest: {
      anchor: '.text-landing',
      bind_extracted: 'append',
      extract: 'self',
      anchor_element: 'item'
    }
  }
});

shortcodes.register('text', {
  anchor: '.shortcode-landing',
  bind_fn: 'append',
  template: 'section.text',
  callback: function(template) {
    const pres = template.querySelectorAll('.text-landing pre')
    for (let i = 0; i < pres.length; i++) {
      pres[i].innerHTML = '' + pres[i].innerHTML + ''
    }
  },
  elements: {
    rest: {
      anchor: '.text-landing',
      bind_extracted: 'append',
      extract: 'self',
      anchor_element: 'template'
    }
  }
});
